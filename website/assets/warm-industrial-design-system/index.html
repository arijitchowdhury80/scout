// then toggle is-in on enter:
const io = new IntersectionObserver(es =>
  es.forEach(e => {
    if (e.isIntersecting) {
      e.target.classList.add('is-in');
      io.unobserve(e.target);
    }
  }), { threshold: .12,
    rootMargin: '0px 0px -8% 0px' });

document.querySelectorAll('.wi-fade')
  .forEach(el => io.observe(el));
